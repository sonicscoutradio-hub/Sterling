      ));

      document.getElementById('setup-banner').classList.remove('visible');
      setStatus('ok', 'Google Sheets connected');
      var resultEl = document.getElementById('url-test-result');
      resultEl.style.color = 'var(--text-muted)';
      resultEl.textContent = '↻ Sending test ping…';

      // Use no-cors — we can't read the response body, but a network error
      // means the URL is wrong; a resolved promise means the script is reachable.
      fetch(url + '?action=ping', { mode: 'no-cors' })
        .then(function() {
          resultEl.style.color = 'var(--accent2)';
          resultEl.textContent = '✅ URL is reachable — Google Sheets is connected! Submit a test entry to confirm data is arriving in your sheet.';
          setStatus('ok', 'Google Sheets connected');
        })
        .catch(function() {
          resultEl.style.color = 'var(--danger)';
          resultEl.textContent = '❌ Could not reach that URL. Double-check the deployment URL and make sure access is set to "Anyone".';
          setStatus('err', 'Unreachable');
        });
    }

    // ===== COPY SCRIPT =====
    function copyScript() {
      var code = document.getElementById('apps-script-code').textContent;
      navigator.clipboard.writeText(code).then(function() {
        showToast('✅ Script copied to clipboard!', 'success');
      }).catch(function() {
        showToast('Select the code above and copy manually.', 'error');
      });
    }

    // ===== TOAST =====
    var toastTimer;
    function showToast(msg, type) {
      var el = document.getElementById('toast');
      el.textContent = msg;
      el.className = 'show ' + (type || '');
      clearTimeout(toastTimer);
      toastTimer = setTimeout(function() { el.className = ''; }, 4000);
    }
